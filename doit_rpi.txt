
#!/bin/bash

if [ "$#" -eq 0 ];then
	echo Give the main file name as argument in order to compile
	exit 0
fi

################# MASTER POSITIONING EXAMPLE (general)
if [ ”$1” = ”positioning” ];then
	#
	echo Compiling general positioning program for RPI. Difference: copied positioning.cpp and vopos.cpp and removed drawing and imshow functions.
	g++ -std=c++11 -fvisibility=hidden -ggdb \
examples/positioning_rpi.cpp \
/usr/lib/arm-linux-gnueabihf/libboost_program_options.a \
`pkg-config --cflags --libs opencv` \
src/vopos_rpi.cpp \
src/angulation.cpp \
src/azipe.cpp \
src/opticalFlow.cpp \
src/homographyVO.cpp \
src/martonRobust.cpp \
/usr/lib/arm-linux-gnueabihf/libgsl.a \
/usr/lib/arm-linux-gnueabihf/libgslcblas.a \
include/Quartic-master/quartic.cpp \
src/videoStream.cpp \
src/dataStream.cpp \
src/logger.cpp \
-o bin/positioning.out

	echo Run with command ./bin/positioning.out [option]
fi

# Kompileringskommando ändrat för raspberry. Static library (.a) file are not at same location as on mac. Found them by running
# find / -name libboost_program_options.a
# and replaced paths in compilation command. The linker had problems because the order was wrong. Must use functions before inculuded
# so wrote them in another order that works.

echo done
